Replace the current clipboard copy line with:
```swift
// Apply AI cleanup if enabled
var finalText = text
if aiProcessingEnabled {
    aiProcessing = true
    defer { aiProcessing = false }
    do {
        finalText = try await llmProvider.process(text)
        NSLog("[OptionC] AI cleanup applied")
    } catch {
        NSLog("[OptionC] AI cleanup skipped: \(error)")
        // finalText stays as text — pipeline continues unchanged
    }
}

// Copy to clipboard
try ClipboardManager.copy(finalText)
```

CRITICAL: Use `defer { aiProcessing = false }` immediately after setting `aiProcessing = true`. This ensures the flag resets on ALL exit paths including Task cancellation. See Research Pitfall 3.
